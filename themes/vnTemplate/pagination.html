function getPaginationHTML(currentPage, totalPages, basePath = '') { // Ensure basePath ends with a slash if (basePath && !basePath.endsWith('/')) { basePath += '/'; } // Don't show pagination if only 1 page if (totalPages <= 1) return ''; // Helper function to generate page URL const getPageUrl = (page) => { if (page === 1) return basePath; return `${basePath}page-${page}/`; }; let html = '
'; // Previous button if (currentPage > 1) { html += `‹ Prev`; } // First page and ellipsis if (currentPage > 4) { html += `1`; if (currentPage > 5) { html += '…'; } } // Page numbers const startPage = Math.max(1, Math.min(currentPage - 3, totalPages - 6)); const endPage = Math.min(totalPages, startPage + 6); for (let page = startPage; page <= endPage; page++) { if (page === currentPage) { html += `${page}`; } else { html += `${page}`; } } // Last page and ellipsis if (currentPage < totalPages - 3) { if (currentPage < totalPages - 4) { html += '…'; } html += `${totalPages}`; } // Next button if (currentPage < totalPages) { html += `Next ›`; } html += '
'; return html; }
